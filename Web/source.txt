	function getAppValue(key) {
		var cnt = 0;
		var isThere = true;
		var frameObj = parent;
		while(!frameObj.FR_TOPFIXED) {
			if(cnt > 20) {

				break;
			}
			frameObj = frameObj.parent;		
			cnt++;
		}
		if(isThere && frameObj.FR_TOPFIXED.topFixedLoaded) {
			return frameObj.FR_TOPFIXED.getAppValueImpl(key);
		} else {
			var url = '/EP/web/common/CM_GetApplicationValue.jsp?key='+key;
			var appValue = '';

				method: 'get',
				asynchronous: false,
				onSuccess: function(request) {
					appValue = lineTrim(request.responseText);		
				}
			});
			return appValue;
		}
	}
	
	function cmCheckSpecialChar( str ) {
	    var rex = /[\!\@\#\$\%\^\&\*\(\)\-\_\=\+\|\\\{\}\[\]\"\'\:\;\<\>\,\.\/\?]/;
	    return str.match(rex);
	}
	
	function cmCheckEmail( str ) {
	    var rex = /^[\w-]+(\.[\w-]+)*@([\w-]+\.)+[a-zA-Z]{2,7}$/;

	}
	
	function cmCheckStrLen(target, maxLen, fieldName) {
	    var temp;
	    var code;
	    var curLen = 0;
	
	    if (typeof target == "undefined") {
	        return true;
	    }
	    
	    var vLen = target.value.length;
	
	    tmpstr = "" ;
	
	    if (vLen > 0) {
	      for(var k = 0; k < vLen; k++) {
	         temp = target.value.charAt(k);
	         code = target.value.charCodeAt(k);
	
	         if (code < 0 || code > 127)
	            curLen += 2;
	         else
	            curLen++;
	
	         if (curLen > maxLen) {
	            alert("최대 입력 범위를 초과하였습니다.");
	            target.value = tmpstr;
	            target.focus();
	            return false;
	         } else {
	            tmpstr += temp;
	         }
	      }
	    }
	    return true;
	}
	
	function cmCheckStrNull(target, fieldName) {
	    var strVal = target.value;
	
	    if (cmTrim(strVal) == "") {
	        alert(fieldName + "을(를) 입력하십시오.");
	        target.value = "";
	        target.focus();
	        return true;
	    } else {
	        target.value = cmTrim(strVal);
	        return false;
	    }
	}
	
	function cmCheckSearchName(target) {
	    var temp;
	
	    if (typeof target == "undefined") {
	        return true;
	    }
	
	    var vLen = target.value.length;
	
	    tmpstr = "" ;
	
 	    if (vLen > 0) {
	      for(var k = 0; k < vLen; k++) {
	         temp = target.value.charAt(k);
	
	         if ((temp == "%") || (temp == "_")) {
	            alert("특수문자 %,_ 는 입력할 수 없습니다.");

	            return false;

	         }
	      }
	    }
	    return true;
	}

    function toggleChild(p, n) {

        nLen = p.length;
        n = parseInt(n);

        npoint = 0;
        npoint1 = nLen;

        n2 = n+1

		if(!document.all["ImgChild"+p]) {
			return;
		}
		
        if (document.all["ImgChild"+p].name == 'Y') {

             document.all["ImgChild"+p].name = 'N';
             document.all["ImgChild"+p].src = "/EP/image/minus.gif";

             for (i=1;i<document.all.length ;i++) {

                StrId = document.all[i].id ;

                if (StrId.substr(npoint, npoint1) == p && document.all[i].sDepth == n2) {
                    document.all[StrId].style.display = "";
                }
            }

        } else {
            document.all["ImgChild"+p].name = 'Y';
            document.all["ImgChild"+p].src = "/EP/image/plus.gif";

            for (i=1;i<document.all.length ;i++) {

                StrId = document.all[i].id ;

                if (StrId.substr(npoint, npoint1) == p && document.all[i].sDepth >= n2) {
                    document.all[StrId].style.display = "none";
                    document.all[StrId].name = 'Y';

                    if (document.all["ImgChild"+StrId]) {
                        document.all["ImgChild"+StrId].name = 'Y';
                        document.all["ImgChild"+StrId].src = "/EP/image/plus.gif";
                    }
                }
             }
        }
    }

    function toggleCheck() {
        for (var i=0; i<document.filterList.elements.length; i++) {
            var ele = document.filterList.elements[i];
            if (ele.name.substring(0, 4) == "ATTO") {
                if (ele.checked) {
                    uncheckAll();
                    break;
                } else {
                    checkAll();
                    break;
                }   //if
            }   //if
        }   //for
    }

    function checkAll() {
        for (var i=0; i<document.filterList.elements.length; i++) {
            var ele = document.filterList.elements[i];
            if (ele.name.substring(0, 4) == 'ATTO')
                ele.checked = true;
        }
    }

    function uncheckAll() {
        for (var i=0; i<document.filterList.elements.length; i++) {
            var ele = document.filterList.elements[i];
            if (ele.name.substring(0, 4) == 'ATTO')
                ele.checked = false;
        }
    }

    function deleteBulletins() {
        var sBullidLst = "";
        for (var i=0; i<document.filterList.elements.length; i++) {
            var ele = document.filterList.elements[i];
            if (ele.name.substring(0, 4) == "ATTO") {
                if (ele.checked) {
                    sBullidLst += "|" + ele.name.substr(4, ele.name.length);
                }   //if
            }   //if
        }   //for

        sBullidLst = sBullidLst.substring(1, sBullidLst.length);
        if (sBullidLst == null || sBullidLst == "") {
            alert("선택된 항목이 없습니다.");
            return;
        } else {
            ans = confirm ("선택된 게시물을 삭제하시겠습니까? \n답변을 가진 게시물은 답변과 함께 삭제됩니다.")
            if (ans) {
                document.filterList.bullidlst.value = sBullidLst;
                document.filterList.action = "./BB_BullDel.jsp";
                document.filterList.selectnum.value = document.filterList.searchcondition.selectedIndex;
                document.all.filterList.submit();
            }else{
                return;
            }//if
        }//if
    }

    function scChanged() {

        var frm = document.filterList;
        var idx = frm.searchcondition.selectedIndex;

        if (frm.searchcondition.value == "CATEGORYID") {
            frm.reply.style.display = "none";
            frm.searchname.style.display = "none";
            frm.category.style.display = "";
            for (i=0;i< frm.length;i++) {
                if (frm.item(i).name.substring(0, 5)=="CODE_") {
                    frm.item(i).style.display = "none";
                }
            }

        } else if (frm.searchcondition[idx].id.substring(0, 5) == "CODE_") {
            frm.reply.style.display = "none";
            frm.searchname.style.display = "none";
            frm.category.style.display = "none";

            for (i=0;i< frm.length;i++) {
                if (frm.item(i).name.substring(0, 5)=="CODE_") {
                    if (frm.item(i).name == frm.searchcondition[idx].id) {
                        frm.item(i).style.display = "";
                    } else {
                        frm.item(i).style.display = "none";
                    }
                }
            }

        } else if (frm.searchcondition.value == "REPLY") {
            frm.reply.style.display = "";
            frm.searchname.style.display = "none";
            frm.category.style.display = "none";
            for (i=0;i< frm.length;i++) {
                if (frm.item(i).name.substring(0, 5)=="CODE_") {
                    frm.item(i).style.display = "none";
                }
            }
        } else {
            frm.reply.style.display = "none";
            frm.searchname.style.display = "";
            frm.category.style.display = "none";
            frm.category[0].selected = true;
            for (i=0;i< frm.length;i++) {
                if (frm.item(i).name.substring(0, 5)=="CODE_") {
                    frm.item(i).style.display = "none";
                }
            }
        }
    }
